    private void btnsubmitActionPerformed(java.awt.event.ActionEvent evt) {                                          

        // long Aadharno = Integer.parseInt(textaadharno.getText());
        String Aadharno = textaadharno.getText();
        String Name = textname.getText();
        int Mobailno = Integer.parseInt(textmobailno.getText());
        String Address = textaddress.getText();

        try {

            pst = con.prepareStatement("insert into electricitybill(aadharno,name,mobailno,address) values(?,?,?,?)");

            pst.setString(1, Aadharno);
            pst.setString(2, Name);
            pst.setInt(3, Mobailno);
            pst.setString(4, Address);

            lbluseraddresult.setText("Add User SuccessFully !");
            pst.executeUpdate();
            pst.close();

        } catch (SQLException ae) {

            lbluseraddresult.setText("Error"+ae);
            System.out.println(ae);
        }

    }                                         

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         

        String PaidAadharno = textaadharno1.getText();
        String PaidName = textname1.getText();
        int PaidUnits = Integer.parseInt(textunits1.getText());

        int Totalbill = PaidUnits * 4;

        try {

            pst = con.prepareStatement("insert into paidelectricitybill(aadharno,name,units,totalbill) values(?,?,?,?)");

            pst.setString(1, PaidAadharno);
            pst.setString(2, PaidName);
            pst.setInt(3, PaidUnits);
            pst.setInt(4, Totalbill);

            lbltotalbillresult.setText("Total BIll :" + Totalbill);
            lblpaybillresult.setText("Bill Paid SuccessFully !");
            pst.executeUpdate();
            pst.close();

        } catch (SQLException ae) {

            lbluseraddresult.setText("Error"+ae);
        }


    }
